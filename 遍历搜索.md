### 栈方法
1. [岛屿的最大面积](https://leetcode.cn/problems/max-area-of-island/description/)
```C++ 
    int maxAreaOfIsland(vector<vector<int>>& grid) {
        // 深度优先遍历，使用栈进行深度遍历
        int mx=0;
        int m=grid.size();
        int n=grid[0].size();
        stack<pair<int,int>> land;        
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j]==1)
                {
                    grid[i][j]=0;
                    land.push({i,j});
                    int area=1;
                    while(!land.empty())
                    {
                        pair<int,int> top=land.top();
                        bool flag=false;
                        for(int k=0;k<4;k++)
                        {
                            int x=top.first+v[k];
                            int y=top.second+h[k];
                            if(x>=0&x<m&&y>=0&&y<n&&grid[x][y]==1)
                            {
                                grid[x][y]=0;
                                land.push({x,y});
                                ++area;
                                flag=true;
                                break; // 跳出for循环
                            }
                        }
                        if(!flag) // flag依旧返回false，表示top周围没有未访问过的陆地
                        {
                            land.pop(); // 此时需要退栈
                        }
                    }
                    mx=max(mx,area);

                }
            }
        }
        return mx;
    }
```

1. [岛屿的最大面积](https://leetcode.cn/problems/max-area-of-island/description/)
```C++
    vector<int> v={0,-1,0,1};
    vector<int> h={1,0,-1,0};
    int m,n; //大小
    int maxAreaOfIsland(vector<vector<int>>& grid) 
    {
        m=grid.size();
        n=grid[0].size();
        int mx=0;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j]==1)
                {
                    int area=1;
                    grid[i][j]=0;
                    area+=dfs(grid,i,j);
                    mx=max(mx,area);
                }
            }
        }
        return mx;
    }
    int dfs(vector<vector<int>>& grid,int i,int j)
    {
        int area=0;
        for(int k=0;k<4;k++)
        {
            int x=i+v[k];
            int y=j+h[k];
            if(x>=0&x<m&&y>=0&&y<n&&grid[x][y]==1)
            {
                ++area;
                grid[x][y]=0;
                area+=dfs(grid,x,y);
            }
        }
        return area;
    }
```

```C++
#include <iostream>
#include <vector>
using namespace std;
vector<pair<int,int>> S;
int n,m;
int x[4]={1,0,-1,0};
int y[4]={0,1,0,-1};

bool dfs(vector<vector<int>> &dp,int i,int j);
int main() {
    cin>>n>>m;
    vector<vector<int>> dp(n,vector<int>(m,0));
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            cin>>dp[i][j];
        }
    }
    dfs(dp,0,0);
    for(auto it=S.begin();it!=S.end();it++)
        cout<<'('<<it->first<<','<<it->second<<')'<<endl;
    return 0;
}
bool dfs(vector<vector<int>> &dp,int i,int j)
{
    S.push_back(make_pair(i,j));
    if(i==n-1&&j==m-1)
        return true;
    dp[i][j]=1;
    bool flag=false;
    for(int k=0;k<4;k++)
    {
        if(reachable(dp,i+y[k],j+x[k]))
        {
            if(dfs(dp,i+y[k],j+x[k]))
                flag=true;
        }  
    }
    if(flag==false)
    {
        S.pop_back();
    }
    return flag;
}
bool reachable(vector<vector<int>> &dp,int i,int j)
{
    if(i>=0&&i<n&&j>=0&&j<m&&dp[i][j]==0)
        return true;
    else
        return false;
}
```

3.[太平洋大西洋水流问题](https://leetcode.cn/problems/pacific-atlantic-water-flow/description/)
```C++
    vector<int> v={0,-1,0,1};
    vector<int> h={1,0,-1,0};
    // 求同时能流向太平洋和大西洋的点
    vector<vector<int>> pacificAtlantic(vector<vector<int>>& heights) {
        vector<vector<int>> res;
        int m=heights.size();
        int n=heights[0].size();
        vector<vector<bool>> reach_p(m,vector<bool> (n,false));
        vector<vector<bool>> reach_a(m,vector<bool> (n,false));
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i==0||j==0)
                {
                    reach_p[i][j]=true;
                    dfs(heights,reach_p,i,j);
                }
                if(i==m-1||j==n-1)
                {
                    reach_a[i][j]=true;
                    dfs(heights,reach_a,i,j);
                }
            }
        }
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(reach_a[i][j]&&reach_p[i][j]) // 该结点能过同时到达太平洋和大西洋
                {
                    res.push_back({i,j});
                }
            }
        }
        return res;
    }
    void dfs(vector<vector<int>>& heights,vector<vector<bool>> &reach,int i,int j)
    {
        for(int k=0;k<4;k++)
        {
            int x=i+v[k];
            int y=j+h[k];
            if(x>=0&&x<heights.size()&&y>=0&&y<heights[0].size())
            {
                if(heights[x][y]>=heights[i][j]&&!reach[x][y]) //reach[x][y]如果为true,说明该点被其他深度优先搜索过，不宜重复
                {
                    reach[x][y]=true; // 该点为更高海拔点
                    dfs(heights,reach,x,y); //从该海拔点继续深度优先
                }
            }
        }
    }
```

## 队列方法 // 对于广度优先遍历，其实使用栈或者队列都可以，如果广度遍历顺序和次序无关的话
1. [岛屿的最大面积](https://leetcode.cn/problems/max-area-of-island/description/)
```C++
   vector<int> v={0,-1,0,1};
    vector<int> h={1,0,-1,0};
    int maxAreaOfIsland(vector<vector<int>>& grid) {
        int mx=0;
        int m=grid.size();
        int n=grid[0].size();
        //stack<pair<int,int>> land;  //使用栈
        queue<pair<int,int>> land;     // 使用队列
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(grid[i][j]==1)
                {
                    land.push({i,j});
                    grid[i][j]=0;
                    int area=1;
                    while(!land.empty())
                    {
                        //pair<int,int> s=land.top(); //使用栈
                        //land.pop();  //使用栈
                        pair<int,int> s=land.front();// 使用队列
                        land.pop(); //使用队列
                        int x=s.first,y=s.second;
                        for(int k=0;k<4;k++)
                        {
                            int x1=x+v[k];
                            int y1=y+h[k];
                            if(x1>=0&&x1<m&&y1>=0&&y1<n&&grid[x1][y1]==1)
                            {
                                ++area;
                                grid[x1][y1]=0;
                                land.push({x1,y1});
                            }
                        }
                    }
                    mx=max(mx,area);
                }
            }
        }
        return mx;
    }
```
