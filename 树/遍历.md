```C++ 
    void  PreOrderTraverse(TreeNode* root)
    {
        if(!root)
            return;
        else{
            visit(root);
            PreOrderTraverse(root->left);
            PreOrderTraverse(root->right);
        }
    }
```
* 栈方法
```C++
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        if(!root)
            return res;
        stack<TreeNode*> s;
        s.push(root);
        while(!s.empty())
        {
            TreeNode *p=s.top();
            res.push_back(p->val);
            s.pop();
            if(p->right)
            {
                s.push(p->right);
            }
            if(p->left)
            {
                s.push(p->left);
            }
        }
        return res;
    }
```

```C++
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        if(!root)
            return res;
        stack<TreeNode*> s;
        s.push(root);
        while(!s.empty())
        {
            TreeNode *p=s.top();
            if(p->left)
            {
                s.push(p->left);
                p->left=nullptr; // 为了防止回溯到p是父结点，防止p又往左子树递归进栈
            }else{
                res.push_back(p->val);
                s.pop();
                if(p->right)
                {
                    s.push(p->right);
                    p->right=nullptr; //可有可无，因为父结点已经出栈，不会往右子树走
                }
            }
        }
        return res;
    }
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        if(!root)
            return res;
        stack<TreeNode*> s; 
        s.push(root);
        while(!s.empty())
        {   
            TreeNode *p;
            while(p=s.top())
                s.push(p->left); //左子树进栈，知道最后进入一个空指针
            s.pop(); //弹出最后进入到空指针
            if(!s.empty()) // 如果弹出一个空指针,此时栈可能为空
            {
                p=s.top(); //访问中结点
                s.pop();
                res.push_back(p->val);
                s.push(p->right); //往右子树走，即使它是空指针，后续被弹出
            }
        }
        return res;
    }
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        if(!root)
            return res;
        stack<TreeNode*> s;
        TreeNode *p=root; //根结点暂时没有进栈
        while(p||!s.empty()) // 注意判断条件为或
        {
            if(p) // 如果p不为空，则进栈
            {
                s.push(p);
                p=p->left;
            }else{
                p=s.top(); //如果p为空，则弹出一个指针赋予p
                s.pop();
                res.push_back(p->val);
                p=p->right; //开始往右子树遍历
            }
        }
        return res;
    }
```
